167        """ Makes a capture to store as the internal threshold value for the SYNC Frame value comparison

183    def sync_time_calibration(self):
184
185        # Clear the system to be in a controlled state
186        for screen in self._screens:
187            screen.clear()
188        
189        self._sync_slice = []
190        for _ in range(__xrandr_Ntry__):
191            # Display and let enough time for the camera to see the pattern
192            img_cap_lst = []
193            uD = self._screens[0]
194            for id_mask, mask in enumerate(uD._sync_mask):
195                uD.set_render_ready(True)
196                uD.displayCUDA(imageData=mask * 255, sync_frame=False)
197                self.clear_FIFO()
198
199                time.sleep(2)
200                img_cap_lst.append(self._img_handler._imgQueue.get()[0])
201            
202            # Post-processing getting the rectangle contour and deduce the coordinates
203            for img in img_cap_lst:
204                # Apply Gaussian blur to reduce noise and improve edge detection
205                blurred = cv2.GaussianBlur((img * 255).astype(np.uint8), (5, 5), 0)
206                
207                # Otsu's thresholding after Gaussian filtering
208                ret, thresh = cv2.threshold(blurred, 50, 255, 0)
209                # Find contours in the edged image 
210                contours, _ = cv2.findContours(thresh, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE) 
211                 
212                # Loop over the contours 
213                for contour in contours:
214                    M = cv2.moments(contour)
215                    if(M["m00"] > 64**2):
216                        # Approximate the contour to a polygon 
217                        epsilon = 0.1 * cv2.arcLength(contour, True) 
218                        approx = cv2.approxPolyDP(contour, epsilon, True) 
219                         
220                        # Check if the approximated contour has 4 points (rectangle) 
221                        if len(approx) == 4: 
222                            x, y, w, h = cv2.boundingRect(contour)
223                            #print(x, y, w, h)
224                            self._sync_slice.append({'x': slice(x, x+w), 'y': slice(y, y+h)})
225        
226            if(len(self._sync_slice) != 4):
227                print("Couldn't find all calibration pattern for camera")
228            else:
229                break

231    def capture_image(self, n_frame=1, sync_frame=False, remove_ref=False):
232        """ Return the copy of the buffer from the threaded method: self._img_handler.OnImageGrabbed()
233        
234        :return: Copy of the acquired image
235        :rtype: np.array([buffer.Height(),  buffer.Width()])
236        """
237        # If the screen havenot been resetted for a while we need to do it
238        if((time.time() - self._reset_time) >= __screen_rst_time__):
239            #print("Screen reset time reached, auto-resetting before capturing new image")
240            self.reset_screen_trigger()
241        
242        n, n_try = 0, 0
243        _img_out = np.zeros((n_frame, 1200, 1920))  # TODO remove magick number
244        _cam_trig_delta = np.zeros(n_frame)
245
246        self.clear_FIFO()
247        self._camera.TimestampLatch.Execute()
248        self._cam_time_start = self._camera.TimestampLatchValue.Value
249        self.trigger_screens()
250        
251        while(n < n_frame):
252        
253            if(not sync_frame):
254                _img_out[n] = self._img_handler._imgQueue.get()[0]
255                n = n + 1
256
257            else:              
258                screen_reset = True
259                # Browse through the data until we have a frame (skipping empty images)
260                for _ in range(__queue_length__):
261                    
262                    _img_out[n], trig_time = self._img_handler._imgQueue.get()
263                    # Maybe make a tiny  function for the following line?
264                    n_slice = n % len(self._sync_slice)
265                    trig_sum = np.sum(_img_out[n][self._sync_slice[n_slice]['y'], self._sync_slice[n_slice]['x']])
266                    
267                    if(trig_sum > self._sync_threshold[n_slice]):
268    
269                        # Check that the frame is coming from after 10ms after the trigger of the screens
270                        delta_t_frame = trig_time - self._cam_time_start
271                        if(delta_t_frame > 9e7):
272                            screen_reset = False
273                            _cam_trig_delta[n] = delta_t_frame
274                            break
275
276                # Check that we dont have a duplicate!
277                if((n_frame > 1) and (n+1 == n_frame) and (np.diff(_cam_trig_delta/1e6).std() > 1)):
278                    #print("Issue with synchronisation, re-acquiring")
279                    screen_reset = True
280                    
281                if(screen_reset):
282                    n = 0
283                    n_try += 1
284                    # If after __detect_sync_frame__ trials we didnot get anything we reset the screen
285                    # We make a new capture which suspend the screen and reset its queue then we reload
286                    # the Cuda Tensor onto the C++ DisplayGL wrapper to retry
287                    if(n_try > __xrandr_Ntry__):
288                        #print("Resetting screens...")
289                        self.reset_screen_trigger()
290                        n_try = 0
291                        
292                    # When everything is ready, we get a capture for the ref of the Camera
293                    self.clear_FIFO()
294                    self._camera.TimestampLatch.Execute()
295                    self._cam_time_start = self._camera.TimestampLatchValue.Value
296                    self.trigger_screens()
297                else:
298                    n = n + 1
299                        
300        self._img_out = _img_out
301        if(remove_ref):
302            for k, _ in enumerate(_img_out):
303                _img_out[k] = cv2.subtract(_img_out[k], self._img_ref)
304        
305        # TODO return maybe an iterator here TODO
306        if(n_frame == 1):
307            return _img_out[0]
308        else:
309            return _img_out

330        
331        :return: None
332        """

338
339        # Have you tried turning it off/on again? (The IT Crowd)
340        time.sleep(0.25)
341        subprocess.run(["xrandr", "--output", "DP-1", "--off", "--nograb",
342                                  "--output", "DP-2", "--off", "--nograb"])
343        for n_try in range(__xrandr_Ntry__):
344            time.sleep(1)
345            sub_proc = subprocess.run(["xrandr", "--output", "DP-6", "--mode", "2560x1440",
346                                       "--output", "DP-2", "--mode", "1920x1080",  "--right-of", "DP-6", "--nograb",
347                                       "--output", "DP-1", "--mode", "1920x1200", "--right-of", "DP-2", "--nograb"])
348            try:
349                sub_proc.check_returncode()
350            except subprocess.CalledProcessError:
351                print("Error while restarting the screen")
352                print("Attempt:", n_try)
353                continue
354
355            else:
356                break
357
358            finally:
359                pass
360        
361        # When the screen is back, we can restore and render on the window on its corresponding screen as before
362        time.sleep(1)
363        self._screens[0].restore_window()
364        time.sleep(0.1)
365        self._screens[0].start_render()
366
367        # We also reset the counter for autoreset
368        self._reset_time = time.time()
369        
370        # When everything is ready, we get a capture for the ref of the Camera
371        if(sync_time):
372            self.sync_time_calibration()
373        self.capture_ref()
